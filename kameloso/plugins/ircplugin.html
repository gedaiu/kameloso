    Module kameloso.plugins.ircplugin

    Contains the definition of an IRCPlugin, as well as a mixin IRCPluginImpl to fully implement one.

    Event handlers can then be module-level functions, annotated with dialect.defs.IRCEvent.Types.

    Example

    import kameloso.plugins.ircplugin;
import kameloso.plugins.common;
import kameloso.plugins.awareness;

@(IRCEvent.Type.CHAN)
@(ChannelPolicy.home)
@(PrefixPolicy.prefixed)
@BotCommand(PrivilegeLevel.anyone, "foo")
void onFoo(FooPlugin plugin, const IRCEvent event)
{
    // ...
}

mixin UserAwareness;
mixin ChannelAwareness;

final class FooPlugin : IRCPlugin
{
    // ...

    mixin IRCPluginImpl;
}
